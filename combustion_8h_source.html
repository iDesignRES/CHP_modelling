3void calculate_fuel_combustion_properties(flow fuel, object &prop) {

7
8 size_t H2O = index_species(fuel.k, "H2O");

11 if (fuel.prop_data == "solid_fuel") {
12 prop.fval_p("n_H", (fuel.i[H].Y / 1) / (fuel.i[C].Y / 12));
13 prop.fval_p("n_O", (fuel.i[O].Y / 16) / (fuel.i[C].Y / 12));
14 prop.fval_p("W_CHxOy", 0.012 + prop.fp("n_H") * 0.001 + prop.fp("n_O") * 0.016);
15 prop.fval_p("nu_O2", 1 + prop.fp("n_H") / 4 - prop.fp("n_O") / 2);
16 prop.fval_p("nu_CO2", 1.0);
17 prop.fval_p("nu_H2O", prop.fp("n_H") / 2);
18 prop.fval_p("V_stoich", fuel.F.M * (1 - fuel.k[H2O].Y) * (1 - fuel.k[ash].Y) *
19 (prop.fp("nu_O2") / 0.21) * (0.02241 / prop.fp("W_CHxOy")));
20 prop.fval_p("vn_CO2_m",
21 (1 - fuel.k[H2O].Y) * prop.fp("nu_CO2") * (0.02241 / prop.fp("W_CHxOy")));
22 prop.fval_p("vn_H2O_m",
23 (1 - fuel.k[H2O].Y) * prop.fp("nu_H2O") * (0.02241 / prop.fp("W_CHxOy")) +
24 fuel.k[H2O].Y * (0.02241 / 0.018));
25 }
26
27 if (fuel.prop_data == "gas_fuel") {
28 size_t CO2 = index_species(fuel.j, "CO2");
29
30 if (CO2 > 0 && fuel.j[CO2].Y > 0) {
31 // gas fuel contains CO2
32 double Y_CO2 = fuel.j[CO2].Y;
33 for (size_t nj = 0; nj < fuel.j.size(); nj++) {
34 if (fuel.j[nj].id != "CO2") {
35 fuel.j[nj].Y = fuel.j[nj].Y / (1.0 - fuel.j[CO2].Y);
36 }
37 }
38 fuel.j[CO2].Y = 0.0;
39 fuel.molec_def = "Y";
40 fuel.calculate_flow_composition();
41 fuel.interpret_molecules();
42
43 prop.fval_p("n_H", (fuel.i[H].Y / 1) / ((fuel.i[C].Y - (Y_CO2 * 12 / 44)) / 12));
44 prop.fval_p("n_O", ((fuel.i[O].Y - (2 * Y_CO2 * 16 / 44)) / 16) /
45 ((fuel.i[C].Y - (Y_CO2 * 12 / 44)) / 12));
46 prop.fval_p("W_CHxOy", 0.012 + prop.fp("n_H") * 0.001 + prop.fp("n_O") * 0.016);
47 prop.fval_p("nu_O2", 1 + prop.fp("n_H") / 4 - prop.fp("n_O") / 2);
48 prop.fval_p("nu_CO2", 1.0);
49 prop.fval_p("nu_H2O", prop.fp("n_H") / 2);
50 prop.fval_p("V_stoich", fuel.F.M * (1.0 - Y_CO2) * (1 - fuel.k[H2O].Y) *
51 (1 - fuel.k[ash].Y) * (prop.fp("nu_O2") / 0.21) *
52 (0.02241 / prop.fp("W_CHxOy")));
53 prop.fval_p("vn_CO2_m", (1 - fuel.k[H2O].Y) * ((1.0 - Y_CO2) * prop.fp("nu_CO2") *
54 (0.02241 / prop.fp("W_CHxOy")) +
55 Y_CO2 * (0.02241 / 0.044)));
56 prop.fval_p("vn_H2O_m", (1 - fuel.k[H2O].Y) * (1.0 - Y_CO2) * prop.fp("nu_H2O") *
57 (0.02241 / prop.fp("W_CHxOy")) +
58 Y_CO2 * (0.02241 / 0.018));
59
60 }

62 else {
63 // otherwise
64
65 prop.fval_p("n_H", (fuel.i[H].Y / 1) / (fuel.i[C].Y / 12));
66 prop.fval_p("n_O", (fuel.i[O].Y / 16) / (fuel.i[C].Y / 12));
67 prop.fval_p("W_CHxOy", 0.012 + prop.fp("n_H") * 0.001 + prop.fp("n_O") * 0.016);
68 prop.fval_p("nu_O2", 1 + prop.fp("n_H") / 4 - prop.fp("n_O") / 2);
69 prop.fval_p("nu_CO2", 1.0);
70 prop.fval_p("nu_H2O", prop.fp("n_H") / 2);
71 prop.fval_p("V_stoich", fuel.F.M * (1 - fuel.k[H2O].Y) * (1 - fuel.k[ash].Y) *
72 (prop.fp("nu_O2") / 0.21) *
73 (0.02241 / prop.fp("W_CHxOy")));
74 prop.fval_p("vn_CO2_m",
75 (1 - fuel.k[H2O].Y) * prop.fp("nu_CO2") * (0.02241 / prop.fp("W_CHxOy")));
76 prop.fval_p("vn_H2O_m",
77 (1 - fuel.k[H2O].Y) * prop.fp("nu_H2O") * (0.02241 / prop.fp("W_CHxOy")) +
78 fuel.k[H2O].Y * (0.02241 / 0.018));
79 }
