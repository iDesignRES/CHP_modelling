3void calculate_fuel_combustion_properties(flow fuel, object &prop) {
4 size_t C = index_species(fuel.i, "C");
5 size_t H = index_species(fuel.i, "H");
6 size_t O = index_species(fuel.i, "O");
7
8 size_t H2O = index_species(fuel.k, "H2O");
9 size_t ash = index_species(fuel.k, "ash");
10
11 if (fuel.prop_data == "solid_fuel") {
12 prop.fval_p("n_H", (fuel.i[H].Y / 1) / (fuel.i[C].Y / 12));
13 prop.fval_p("n_O", (fuel.i[O].Y / 16) / (fuel.i[C].Y / 12));
14 prop.fval_p("W_CHxOy",
15 0.012 + prop.fp("n_H") * 0.001 + prop.fp("n_O") * 0.016);
16 prop.fval_p("nu_O2", 1 + prop.fp("n_H") / 4 - prop.fp("n_O") / 2);
17 prop.fval_p("nu_CO2", 1.0);
18 prop.fval_p("nu_H2O", prop.fp("n_H") / 2);
19 prop.fval_p("V_stoich",
20 fuel.F.M * (1 - fuel.k[H2O].Y) * (1 - fuel.k[ash].Y) *
21 (prop.fp("nu_O2") / 0.21) * (0.02241 / prop.fp("W_CHxOy")));
22 prop.fval_p("vn_CO2_m", (1 - fuel.k[H2O].Y) * prop.fp("nu_CO2") *
23 (0.02241 / prop.fp("W_CHxOy")));
24 prop.fval_p("vn_H2O_m", (1 - fuel.k[H2O].Y) * prop.fp("nu_H2O") *
25 (0.02241 / prop.fp("W_CHxOy")) +
26 fuel.k[H2O].Y * (0.02241 / 0.018));
27 }
28}
