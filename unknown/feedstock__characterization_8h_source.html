205 // -1.53; // Milne's formulae (from Phyllis) kC = 34.8; kH = 93.9; kO =
206 // -10.8; kS = 10.5; kN = 6.3; kH2O = -2.45; kA = 0.0; // Ref. Christensen, T.

217 } else if (feed.i[n].id == "N") {

224 // fetching the proximates:
225 for (size_t n = 0; n < feed.k.size(); n++) {
226 if (feed.k[n].id == "ash") {
227 yA = feed.k[n].Y;
228 } else if (feed.k[n].id == "H2O") {
229 yH2O = feed.k[n].Y;

231 }
232
233 if (yH2O > 1)
234 yH2O = yH2O / 100;

239 // yN / sum_y;

242 if (feed.P.LHV_dry == 0) {
