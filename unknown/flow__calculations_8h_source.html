24void flow::calculate_flow_composition() {
25 double sum_Y, sum_X, sum_X_MW, sum_Y_MW;
26 P.MW = 0;
27 if (j.size() == 1) {
28 P.MW = j[0].P.MW;
29 j[0].Y = 1.0;
30 j[0].X = 1.0;
31 }
32
33 if (j.size() > 1) {
34 sum_Y = 0.0;
35 for (size_t n = 0; n < j.size(); n++) {
36 sum_Y += j[n].Y;
37 }
38 sum_X = 0.0;
39 for (size_t n = 0; n < j.size(); n++) {
40 sum_X += j[n].X;
41 }
42 if (molec_def != "Y" && sum_Y > 0.0) {
43 molec_def = "Y";
44 }
45 if (molec_def != "X" && sum_Y == 0.0 && sum_X > 1e-6) {
46 molec_def = "X";
47 }
48
49 if (molec_def == "Y") {
50 sum_Y_MW = 0.0;
51 for (size_t n = 0; n < j.size(); n++) {
52 if (j[n].Y > 0 && j[n].P.MW > 0) {
53 sum_Y_MW += j[n].Y / j[n].P.MW;
54 }
55 }
56 for (size_t n = 0; n < j.size(); n++) {
57 if (j[n].Y > 0 && j[n].P.MW > 0 && sum_Y_MW > 0) {
58 j[n].X = (j[n].Y / j[n].P.MW) / sum_Y_MW;
59 }
60 if (j[n].Y == 0) {
61 j[n].X = 0;
62 }
63 }
64 } else if (molec_def == "X") {
65 sum_X_MW = 0.0;
66 for (size_t n = 0; n < j.size(); n++) {
67 if (j[n].X > 0 && j[n].P.MW > 0) {
68 sum_X_MW += j[n].X * j[n].P.MW;
69 }
70 }
71 for (size_t n = 0; n < j.size(); n++) {
72 if (j[n].X > 0 && j[n].P.MW > 0 && sum_X_MW > 0) {
73 j[n].Y = (j[n].X * j[n].P.MW) / sum_X_MW;
74 }
75 if (j[n].X == 0) {
76 j[n].Y = 0;
77 }
78 }
79 }
80
81 for (size_t n = 0; n < j.size(); n++) {
82 P.MW += j[n].X * j[n].P.MW;
83 }
84 }

87void flow::calculate_flow_properties(string state_def) {
88 double cp, hVap, BP, DP;
89 cp = 0.0;
90 hVap = 0.0;
91 BP = 0.0;
92 DP = 0.0;
93 P.cp = 0;
94 P.rho = 0;
95
96 if (prop_data != "refprop" && prop_data != "NASA" && prop_data != "solid_fuel" &&
97 prop_data != "NIST") {
98 define_flow_prop_data();
99 }
100
101 // cout << "flow prop_data: " << prop_data << endl;
102
103 if (prop_data == "solid_fuel") {
104 calculate_solid_fuel();
105 }
106
107 if (prop_data == "NIST") {
108
109 calculate_species_properties(state_def);
110
111 P.cp = 0.0;
112 P.h = 0.0;
113 P.s = 0.0;
114
115 for (size_t n = 0; n < j.size(); n++) {
116 P.cp += j[n].Y * j[n].P.cp;
117 P.h += j[n].Y * j[n].P.h;
118 P.s += j[n].Y * j[n].P.s;
119 }
120
121 P.ht = P.cp * (F.T - 25.0);
122 }
123
124 if (prop_data == "hybrid") {
125
126 calculate_species_properties(state_def);
127
128 if (j.size() == 1) {
129 P.cp = j[0].P.cp;
130 P.rho = j[0].P.rho;
131 P.h = j[0].P.h;
132 P.ht = j[0].P.cp * (F.T - 25.0);
133 if (P.h == 0 && P.ht > 0 && P.hf == 0) {
134 j[0].P.h = j[0].P.ht;
135 }
136 P.s = j[0].P.s;
137 P.g = j[0].P.g;
138 }
139
140 if (j.size() > 1) {
141 for (size_t n = 0; n < j.size(); n++) {
142 // update flow properties (assume mass basis at first (MUST BE CHANGED LATER)):
143 // P.cp += j[n].X * j[n].P.cp;
144 P.cp += j[n].Y * j[n].P.cp;
145 P.h += j[n].Y * j[n].P.h;
146 P.ht += j[n].Y * j[n].P.ht;
147 P.hf += j[n].Y * j[n].P.hf;
148 P.s += j[n].Y * j[n].P.s;
149 P.g += j[n].Y * j[n].P.g;
150 P.hVap += j[n].P.hVap * j[n].Y;
151
152 // Bubble point && dew point will be the minimum && maximum for the individual
153 // components
154 if (n == 0) {
155 BP = j[n].P.Tsat;
156 DP = j[n].P.Tsat;
157 }
158
159 if (j[n].P.Tsat < BP) {
160 BP = j[n].P.Tsat;
161 }
162
163 if (j[n].P.Tsat > DP) {
164 DP = j[n].P.Tsat;
165 }
166 }
167
168 double sum_Y_rho = 0.0;
169 for (size_t n = 0; n < j.size(); n++) {
170 sum_Y_rho += j[n].Y / j[n].P.rho;
171 }
172 P.rho = 1.0 / sum_Y_rho;
173 }
174 }

273void flow::calculate_flow_parameters() {
274 if (F.M == 0 && F.VN > 0 && F.N > 0 && P.MW > 0) {
275 F.M = F.N * P.MW;
276 } else if (F.N == 0 && F.VN > 0 && F.M > 0 && P.MW > 0) {
277 F.N = F.M / P.MW;
278 } else if (F.VN > 0 && F.M == 0 && F.N == 0 && P.MW > 0) {
279 F.N = F.VN / 0.02214;
280 F.M = F.N * P.MW;
281 }
282
283 F.H = F.M * P.h;
284 F.Ht = F.M * P.ht;
285 F.Hf = F.M * P.hf;
286
287 if (F.H == 0 && F.Ht > 0) {
288 F.H = F.Ht;
289 }
290 if (F.Hf == 0 && F.H > 0 && F.Ht > 0) {
291 F.Hf = F.H - F.Ht;
292 }
