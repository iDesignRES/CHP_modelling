155void flow::calculate_flow_parameters() {
156 if (F.M == 0 && F.VN > 0 && F.N > 0 && P.MW > 0) {

158 } else if (F.N == 0 && F.VN > 0 && F.M > 0 && P.MW > 0) {
159 F.N = F.M / P.MW;
160 } else if (F.VN > 0 && F.M == 0 && F.N == 0 && P.MW > 0) {
161 F.N = F.VN / 0.02214;
162 F.M = F.N * P.MW;
163 }
164
165 F.H = F.M * P.h;
166 F.Ht = F.M * P.ht;
167 F.Hf = F.M * P.hf;
168
169 if (F.H == 0 && F.Ht > 0) {
170 F.H = F.Ht;
171 }
172 if (F.Hf == 0 && F.H > 0 && F.Ht > 0) {
173 F.Hf = F.H - F.Ht;
