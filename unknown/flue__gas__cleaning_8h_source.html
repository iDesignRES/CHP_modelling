9void NOx_reduction_model(flow &in, flow &out, vector<flow> &cons, vector<parameter> &par) {
10 in.print_flow();

14 double X_NOx_clean =

20 in.calculate_flow("PT");
21
22 cons.push_back(flow("ammonia", "NH3_pure"));
23 size_t ammonia = index_flow(cons, "ammonia");
24 cons[ammonia].F.T =
25 atof(get_parameter_value(par, "process", "NOx_REDUCTION", "T_NH3").c_str());
26 cons[ammonia].F.P =
27 atof(get_parameter_value(par, "process", "NOx_REDUCTION", "P_NH3").c_str());
28 cons[ammonia].calculate_flow("PT");
29 cons[ammonia].F.M = cons[ammonia].P.rho * in.F.M * 1e-3 * 1.270 / (2166.8 / 3600.0);
30
31 out = in;
32 out.mix_flows(in, cons[ammonia]);
33 size_t NO2 = index_species(out.j, "NO2");
34 out.j[NO2].X = X_NOx_clean;
35 size_t NH3 = index_species(out.j, "NH3");
36 out.j[NH3].X -= in.j[NO2].X - X_NOx_clean;
37 out.calculate_flow("PT");
38 out.print_flow();
39
40 cout << "fg in M: " << in.F.M << endl;
41 cout << "cons M: " << cons[ammonia].F.M << endl;
42 cout << "cons rho: " << cons[ammonia].P.rho << endl;
43 cout << "fg out M: " << out.F.M << endl;
44
45 export_parameter(par, "output", "process", "NOx_REDUCTION", "M_NH3",
46 to_string(cons[ammonia].F.M));
47}
